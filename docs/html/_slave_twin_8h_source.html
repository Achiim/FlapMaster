14/*
15 Flap Slave Twin Definition
16 --------------------------
17 provides functionality on a flap module level to do:
18 - Flap Drum calibration
19 - check for Hall Sensor presence
20 - show Flap number
21 - move one Flap (next/prev)
22 - move some Steps (next/prev) to set offset steps, if desired flap is not shown after calibration
23 - save parameter to EEPROM on Flap module
24 - step measurement
25 - speed measurement
26 - reset to factory settings
27 - set new I2C address
28 - avilability check
29 - scan for new devices
30 - remote control via IR
31
32*/

44enum TwinCommands {
45 TWIN_NO_COMMAND = 0, // no command
46 TWIN_SHOW_FLAP = 10, // show Flap with this flap number
47 TWIN_CALIBRATION = 20, // do calibration
48 TWIN_STEP_MEASUREMENT = 30, // measure steps for one revolution
49 TWIN_SPEED_MEASUREMENT = 40, // measure speed in ms for one revolution
50 TWIN_SENSOR_CHECK = 50, // check if hall sensor is working
51 TWIN_NEXT_FLAP = 60, // move to next flap
52 TWIN_PREV_FLAP = 70, // move to previous flap
53 TWIN_NEXT_STEP = 80, // move to next step (25 steps) to adjust calibration
54 TWIN_PREV_STEP = 90, // move to previous step (-25 steps) to adjust calibration
55 TWIN_SET_OFFSET = 100, // save calibration offset in EEPROM
56 TWIN_RESET = 110, // do complete factory reset of slave I2C address = 0x55, no serialNumber, EEPROM
57 TWIN_AVAILABILITY = 120, // this command is used to check if twin device is available
58 TWIN_REGISTER = 130, // this command is used to register a twin device
59 TWIN_NEW_ADDRESS = 140 // this command is used to set the base address for the twin
60}; // list of possible twin commands
